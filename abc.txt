Public Sub CopyFieldPropstoTable (ByVal pXDoc As CscXDocument)
   Dim tmpTable As CscXDocTable
   Dim tmpCell As CscXDocTableCell
   Dim tmpLocator As CscXDocFieldAlternatives
   Dim i As Integer
   Dim j As Integer
   Dim DimsCount As Integer
   Dim tmpCount As Integer
   Dim AzlUsed As Boolean
   Dim tmpSubField As String

   'Identify which locator is being utilized
   Set tmpTable = pXDoc.Fields.ItemByName("ExtractedData").Table

   If pXDoc.ExtractionClass = "Other" Then
         Exit Sub
   End If

   If tmpTable.Rows.Count > 0 Then
      For i= 0 To tmpTable.Rows.Count -1
         Set tmpCell = tmpTable.Rows(i).Cells.ItemByIndex(4)
            If InStr(tmpCell.Text, "LIF") <> 0 Then
               If pXDoc.Locators.ItemByName("FL_Liftgate").Alternatives.Count > 0 Then
                     Set tmpLocator = pXDoc.Locators.ItemByName("FL_Liftgate").Alternatives
                  ElseIf pXDoc.Locators.ItemByName("AZ_KeywordCB").Alternatives.Count > 0 Then
                        If pXDoc.Locators.ItemByName("AZ_KeywordCB").Alternatives(0).SubFields.ItemByName("SF_ZoneLiftgate").Text<>"" Then
                           Set tmpLocator = pXDoc.Locators.ItemByName("AZ_KeywordCB").Alternatives
                           AzlUsed =True
                           tmpSubField = "SF_ZoneLiftgate"
                        End If
                  Else
                     tmpCount =1
                  End If
            ElseIf  InStr(tmpCell.Text, "LGO") <> 0 Then
              If pXDoc.Locators.ItemByName("FL_LiftgatePickup").Alternatives.Count > 0 Then
                     Set tmpLocator = pXDoc.Locators.ItemByName("FL_LiftgatePickup").Alternatives
                  ElseIf pXDoc.Locators.ItemByName("AZ_KeywordCB").Alternatives.Count > 0 Then
                        If pXDoc.Locators.ItemByName("AZ_KeywordCB").Alternatives(0).SubFields.ItemByName("SF_ZoneLiftgatePickup").Text<>"" Then
                           Set tmpLocator = pXDoc.Locators.ItemByName("AZ_KeywordCB").Alternatives
                           AzlUsed =True
                           tmpSubField = "SF_ZoneLiftgatePickup"
                        End If
                  Else
                     tmpCount =1
                  End If
            ElseIf  InStr(tmpCell.Text, "ID") <> 0 Then
               If pXDoc.Locators.ItemByName("FL_ID").Alternatives.Count > 0 Then
                     Set tmpLocator = pXDoc.Locators.ItemByName("FL_ID").Alternatives
                  ElseIf pXDoc.Locators.ItemByName("AZ_KeywordCB").Alternatives.Count > 0 Then
                        If pXDoc.Locators.ItemByName("AZ_KeywordCB").Alternatives(0).SubFields.ItemByName("SF_ZoneInsideDelivery").Text<>"" Then
                           Set tmpLocator = pXDoc.Locators.ItemByName("AZ_KeywordCB").Alternatives
                           AzlUsed =True
                           tmpSubField = "SF_ZoneInsideDelivery"
                        End If
                  Else
                     tmpCount =1
                  End If

            ElseIf  InStr(tmpCell.Text, "GTE") <> 0 Then
                  If pXDoc.Locators.ItemByName("FL_GTE").Alternatives.Count > 0 Then
                     Set tmpLocator = pXDoc.Locators.ItemByName("FL_GTE").Alternatives
                  ElseIf pXDoc.Locators.ItemByName("AZ_GD").Alternatives.Count > 0 Then
                        If pXDoc.Locators.ItemByName("AZ_GD").Alternatives(0).SubFields.ItemByName("SF_ZoneCBD").Text<>"" Then
                           Set tmpLocator = pXDoc.Locators.ItemByName("AZ_GD").Alternatives
                           AzlUsed =True
                           tmpSubField = "SF_ZoneCBD"
                        End If
                  Else
                     tmpCount =1
                  End If
            ElseIf  InStr(tmpCell.Text, "GAMD") <> 0 Then
                  If pXDoc.Locators.ItemByName("FL_GAMD").Alternatives.Count > 0 Then
                     Set tmpLocator = pXDoc.Locators.ItemByName("FL_GAMD").Alternatives
                  ElseIf pXDoc.Locators.ItemByName("AZ_GD").Alternatives.Count > 0 Then
                        If pXDoc.Locators.ItemByName("AZ_GD").Alternatives(0).SubFields.ItemByName("SF_ZoneAMDelivery").Text<>"" Then
                           Set tmpLocator = pXDoc.Locators.ItemByName("AZ_GD").Alternatives
                           AzlUsed =True
                           tmpSubField = "SF_ZoneAMDelivery"
                        End If
                  Else
                     tmpCount =1
                  End If

            ElseIf  InStr(tmpCell.Text, "APPT") <> 0 Then
               Set tmpLocator = pXDoc.Locators.ItemByName("FL_Appt").Alternatives
            ElseIf  InStr(tmpCell.Text, "DIMS") <> 0 Then
               Set tmpLocator = pXDoc.Locators.ItemByName("FL_DIMS").Alternatives
               DimsCount = DimsCount + 1
            Else
               tmpCount =1
            End If

         If AzlUsed = True Then
               'Statements here

                  tmpCell.Top = tmpLocator(0).SubFields.ItemByName(tmpSubField).Top
                  tmpCell.Width = tmpLocator(0).SubFields.ItemByName(tmpSubField).Width
                  tmpCell.Height = tmpLocator(0).SubFields.ItemByName(tmpSubField).Height
                  tmpCell.Left = tmpLocator(0).SubFields.ItemByName(tmpSubField).Left
                  tmpCell.PageIndex =tmpLocator(0).SubFields.ItemByName(tmpSubField).PageIndex
                  tmpCell.ExtractionConfident = IIf(tmpLocator(0).SubFields.ItemByName(tmpSubField).Confidence>=.70,True,False)

         Else
            If InStr(tmpCell.Text,"DIMS") <>0 Then
                  j= DimsCount -1
                  tmpCell.Top= tmpLocator(j).Top
                  tmpCell.Width = tmpLocator(j).Width
                  tmpCell.Height = tmpLocator(j).Height
                  tmpCell.Left = tmpLocator(j).Left
                  tmpCell.PageIndex =tmpLocator(j).PageIndex
                  tmpCell.ExtractionConfident = IIf(tmpLocator(j).Confidence>=.70,True,False)

            ElseIf tmpCount = 0 Then
                tmpCell.Top= tmpLocator(0).Top
                tmpCell.Width = tmpLocator(0).Width
                tmpCell.Height = tmpLocator(0).Height
                tmpCell.Left = tmpLocator(0).Left
                tmpCell.PageIndex =tmpLocator(0).PageIndex
                tmpCell.ExtractionConfident = IIf(tmpLocator(0).Confidence>=.70,True,False)

            End If
         End If

      Next
   End If

End Sub
